033package org.jfree.graphics2d.svg;
034
035/**
036 * An object that can (optionally) translate one font family name to an
037 * alternative.  A <code>FontMapper</code> is assigned to an 
038 * {@link SVGGraphics2D} instance.  The default implementation will map
039 * Java logical font names to the equivalent SVG generic font names.
040 * 
041 * @since 1.5
042 */
043public interface FontMapper {
044    
045    /**
046     * Maps the specified font family name to an alternative, or else returns
047     * the same family name.
048     * 
049     * @param family  the font family name (<code>null</code> not permitted).
050     * 
051     * @return The same font family name or an alternative (never 
052     *     <code>null</code>).
053     */
054    String mapFont(String family);
055
056}
