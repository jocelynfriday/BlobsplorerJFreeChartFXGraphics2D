039/**
040 * Defines the rendering hints that can be used with the {@link SVGGraphics2D} 
041 * class.  The supported hints are:<br>
042 * <ul>
043 * <li>{@link #KEY_IMAGE_HANDLING} that controls how images are handled 
044 * (embedded in the SVG, or referenced externally);</li>
045 * <li>{@link #KEY_IMAGE_HREF} that allows the caller to specify the image
046 * href attribute for the next image;</li>
047 * <li>{@link #KEY_TEXT_RENDERING} that allows configuration of the preferred 
048 * value of the SVG <code>text-rendering</code> attribute in text elements;</li>
049 * <li>{@link #KEY_ELEMENT_ID} that allows the caller to specify the element
050 * ID for the next element;</li>
051 * <li>{@link #KEY_BEGIN_GROUP} tells the <code>SVGGraphics2D</code> instance 
052 * to start a new group element with an id equal to the hint value (which must 
053 * be an instance of String).  Any other <code>Graphics2D</code> implementation 
054 * will ignore this hint;</li>
055 * <li>{@link #KEY_END_GROUP} tells the <code>SVGGraphics2D</code> instance 
056 * to end a group element.  The hint value is ignored.  The caller assumes 
057 * responsibility for balancing the number of <code>KEY_BEGIN_GROUP</code> and 
058 * <code>KEY_END_GROUP</code> hints.  Any other <code>Graphics2D</code> 
059 * implementation will ignore this hint.</li>
060 * </ul>
061 * 
062 */
063public final class SVGHints {
064
065    private SVGHints() {
066        // no need to instantiate this

428    private static RenderingHints.Key getOrsonChartsBeginElementKey() {
429        Class<?> renderingHintsClass;

434            return (RenderingHints.Key) f.get(null);

436            return null;

440            return null;

443        } catch (IllegalAccessException ex) {
444            return null;
445        }

488    /**
489     * A key for hints used by the {@link SVGGraphics2D} class.
490     */
491    public static class Key extends RenderingHints.Key {
492
493        /**
494         * Creates a new instance.
495         * 
496         * @param privateKey  the private key. 
497         */
498        public Key(int privateKey) {
499            super(privateKey);    
500        }
501    
502        /**
503         * Returns <code>true</code> if <code>val</code> is a value that is
504         * compatible with this key, and <code>false</code> otherwise.
505         * 
506         * @param val  the value.
507         * 
508         * @return A boolean. 
509         */
510        @Override
511        public boolean isCompatibleValue(Object val) {
512            switch (intKey()) {
513                case 0:
514                    return VALUE_IMAGE_HANDLING_EMBED.equals(val)
515                            || VALUE_IMAGE_HANDLING_REFERENCE.equals(val);
516                case 1:
517                    return VALUE_TEXT_RENDERING_AUTO.equals(val)
518                            || VALUE_TEXT_RENDERING_INHERIT.equals(val)
519                            || VALUE_TEXT_RENDERING_LEGIBILITY.equals(val)
520                            || VALUE_TEXT_RENDERING_PRECISION.equals(val)
521                            || VALUE_TEXT_RENDERING_SPEED.equals(val);
522                case 2: // KEY_IMAGE:URL
523                    return val == null || val instanceof String;
524                case 3: // KEY_ELEMENT_ID
525                    return val == null || val instanceof String;    
526                case 4: // KEY_BEGIN_GROUP
527                    return val == null || val instanceof String;
528                case 5: // KEY_END_GROUP
529                    return true; // the value is ignored
530                case 6: // KEY_ELEMENT_TITLE
531                    return val instanceof String;
532                default:
533                    throw new RuntimeException("Not possible!");
534            }
535        }
536    }
