033package org.jfree.graphics2d.svg;
034
035import java.awt.Font;
036import java.util.HashMap;
037import java.util.Map;
038import org.jfree.graphics2d.Args;
039
040/**
041 * A default implementation of the {@link FontMapper} interface.  This 
042 * implementation will map the Java logical fonts to equivalent SVG generic
043 * fonts.  You can add your own mappings if you need to.
044 * 
045 * @since 1.5
046 */
047public class StandardFontMapper implements FontMapper {
048    
049    /** Storage for the alternates. */
050    private Map<String, String> alternates;
051    
052    /**
053     * Creates a new instance with mappings for the Java logical fonts.
054     */
055    public StandardFontMapper() {
056        this.alternates = new HashMap<String, String>();
057        this.alternates.put(Font.DIALOG, "sans-serif");
058        this.alternates.put(Font.DIALOG_INPUT, "monospace");
059        this.alternates.put(Font.SANS_SERIF, "sans-serif");
060        this.alternates.put(Font.SERIF, "serif");
061        this.alternates.put(Font.MONOSPACED, "monospace");
062    }
063
064    /**
065     * Returns the mapped (alternate) font family name.
066     * 
067     * @param family  the font family (<code>null</code> not permitted).
068     * 
069     * @return The alternate font family name (possibly <code>null</code>). 
070     */
071    public String get(String family) {
072        Args.nullNotPermitted(family, "family");
073        return this.alternates.get(family);
074    }
075    
076    /**
077     * Adds a font family mapping (if the specified alternate is 
078     * <code>null</code> it has the effect of clearing any existing mapping).
079     * 
080     * @param family  the font family name (<code>null</code> not permitted).
081     * @param alternate  the alternate (<code>null</code> permitted).
082     */
083    public void put(String family, String alternate) {
084        Args.nullNotPermitted(family, "family");
085        this.alternates.put(family, alternate);
086    }
087    
088    /**
089     * Maps the specified font family name to an alternative, or else returns
090     * the same family name.
091     * 
092     * @param family  the font family name (<code>null</code> not permitted).
093     * 
094     * @return The same font family name or an alternative (never 
095     *     <code>null</code>).
096     */
097    @Override
098    public String mapFont(String family) {
099        Args.nullNotPermitted(family, "family");
100        String alternate = this.alternates.get(family);
101        if (alternate != null) {
102            return alternate;
103        }
104        return family;
105    }
106
107}
